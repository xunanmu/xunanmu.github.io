二叉树--非递归遍历

前面讲了二叉树的递归遍历,今天来说说非递归遍历

头文件准备

如果你不清楚下面头文件请移步，《C语言封装stack》，《C语言封装Queue》

#include <stdio.h>
#include "Stack.h"
#include "Queue.h"

#define true  (char)1
#define false (char)0
typedef char Status;

typedef struct BinaryTree {
    char data;
    struct BinaryTree *leftChild;
    struct BinaryTree *rightChild;
} binaryTree_size, *BinaryTree;

前序遍历

1.方法一是常规思路，是按照前序先访问父节点 把他压入stack，在访问其左孩子，直到左孩子为Null把其出栈访问他的右孩子；继续重复操作访问其左孩子，直到访问完所有节点。

   void preOrderTraversal_1(BinaryTree T) {
       if (!T)
           return;
       Stack s = initStack();
       do {
           if (T != NULL) {
               printf("%c ", T->data);
               s.push(&s, T);
               T = T->leftChild;
           }
           else {
               T = s.top(&s);
               s.pop(&s);
               T = T->rightChild;
           }
       } while (!s.empty(&s) || T);
   }

2.方法2思路不管左右是否为空通通以满二叉压入栈，首先把根节点压入栈，然后弹出根节点判断他是不是空节点如果是空就不访问，栈空直接循环结束；如果不为空就访问他，并先压入右孩子再压左孩子，有的为什么先压右孩子？。。。。。。因为熟悉栈的同学都知道的栈是先压后出的。

   void preOrderTraversal_2(BinaryTree T) {
       if (!T)    //习惯性敲为Null返回条件，这个判断可以删掉的
           return;  
       Stack s = initStack();
       s.push(&s, T);
       do {
           T = s.top(&s);
           s.pop(&s);
           if (T != NULL) {
               printf("%c ", T->data);
               s.push(&s, T->rightChild);
               s.push(&s, T->leftChild);
           }
       } while (!s.empty(&s));
   }

3.第三个方法和第一个方法刚思路类似只不过处理节点刚好相反。我们可以每次访问该节点的时候把他的右孩子先压入，然后再访问他左孩子，直到他左孩子为Null时弹出他的右孩子继续重复操作访问完所有节点。

   void preOrderTraversal_3(BinaryTree T) {
       if (!T)
           return;
       Stack s = initStack();
       while (T) {
           printf("%c ", T->data);
           if (T->rightChild)
               s.push(&s, T->rightChild);
           if (T->leftChild)
               T = T->leftChild;
           else
               T = s.top(&s),s.pop(&s);
       }
   }

中序遍历

思路很简单：他和前序的方法一结构几乎一样，按照中序思路先访问左孩子，那么我们就把左孩子一直压，直到他为左孩子为空时，把他出栈访问，然后访问他右孩子继续压他右孩子的左孩子；如果没有右孩子就出栈继续重复操作。

void inOrderTraversal(BinaryTree T) {
    if (!T)
        return;
    Stack s = initStack();
    do {
        if (T != NULL) {
            s.push(&s, T);
            T = T->leftChild;
        }
        else {
            T = s.top(&s);
            s.pop(&s);
            printf("%c ", T->data);
            T = T->rightChild;
        }
    } while (!s.empty(&s) || T);
}

注释：你们也可以仿照前序的方法2和方法3来写，这里我就不重复说了，他们结构几乎都一样，只是细节有点小差别。

后序遍历

1.后序遍历时由于访问完左右子树后才能访问根结点，因此需要将根结点在栈内保留到左右子树被访问后，才能访问。我们这里可以利用stack的思路，在父节点这先不访问，把他压入栈，如果他的右孩子不为Null压入栈一直压右孩子，否则压入左孩子。这里单栈不好操作，我们用双栈遍历后序，具体操作，一个栈访问，一个栈压入，栈S1先压入父节点，然后判断左右孩子不为Null压入左右孩子；另一个栈S2压入父节点，然后S1把栈顶弹出（注意：这里栈top已经不是父节点了而是他的子节点），继续重复操作。

void postOrderTraversal_1(BinaryTree T) {
    if (!T)
        return;
    Stack s1 = initStack();
    Stack s2 = initStack();
    do {
        s1.pop(&s1);
        if (T->leftChild)
            s1.push(&s1, T->leftChild);
        if (T->rightChild)
            s1.push(&s1, T->rightChild);
        s2.push(&s2, T);
        T = s1.top(&s1);
    } while (!s1.empty(&s1));

    while (!s2.empty(&s2)) {
        T = s2.top(&s2);
        s2.pop(&s2);
        printf("%c ", T->data);
    }
}

2.首先将整棵二叉树的根结点入栈。取栈顶结点，在这里我们需要在定义一个变量pre代表前面一个访问的结点，若其不存在左右孩子，或存在左孩子或右孩子，但其左子结点和右子结点都被访问过了，则访问此结点，并将他从栈中弹出。若非上述两种情况，则将其的右子结点和左子结点依次入栈。重复上述操作，直到栈为空，遍历结束。

void postOrderTraversal_2(BinaryTree T) {
    if (!T)
        return;
    Stack s = initStack();
    BinaryTree pre=NULL;
    s.push(&s,T);
    while (!s.empty(&s)){
        if (T->leftChild==T->rightChild||pre==T->rightChild||pre==T->leftChild)
        {
            printf("%c ", T->data);
            s.pop(&s);
            pre=T;
        } else{
            if (T->rightChild)
                s.push(&s,T->rightChild);
            if (T->leftChild)
                s.push(&s,T->leftChild);
        }
        T=s.top(&s);
    }
}

3.思路和方法2类似：需要将根结点在栈内保留到左右子树被访问后，我们先一直压他的左孩子，到尾时就访问此节点，然后访问他的右子树。注意：我们同样需要在定义一个变量pre代表前面一个访问的结点，当pre与栈顶结点的右子树结点相同时，则不再将右子树结点压入栈中否则就呵呵哒了！原因就是
：当右子树弹出后遇到根结点又会将右子树结点压入栈中，造成死循环。

void postOrderTraversal_3(BinaryTree T) {
    if (!T)
        return;
    Stack s = initStack();
    BinaryTree pre=NULL;
    s.push(&s,T);
    while (!s.empty(&s))
    {
        while (((BinaryTree)s.top(&s))->leftChild){
            s.push(&s,((BinaryTree)s.top(&s))->leftChild);
        }
        while (!s.empty(&s)){
            if (((BinaryTree)s.top(&s))->rightChild==NULL||
                ((BinaryTree)s.top(&s))->rightChild==pre){
                printf("%c ", ((BinaryTree)s.top(&s))->data);
                pre=(BinaryTree)s.top(&s);
                s.pop(&s);
            }
            else if(((BinaryTree)s.top(&s))->rightChild)
            {
                s.push(&s,((BinaryTree)s.top(&s))->rightChild);
                break;
            }
        }
    }
}

层序遍历

层次就非常简单了，利用queue操作，我们先把根入队，然后把他出队访问，判断他左右孩子不为Null入队，然后重复操作。

void levelTraversal(BinaryTree T) {
    if (!T)
        return;
    Queue q=initQueue();
    q.enqueue(&q,T);
    while (!q.empty(&q)){
        T=q.getfront(&q);
        printf("%c ",T->data);
        if (T->leftChild)
            q.enqueue(&q,T->leftChild);
        if (T->rightChild)
            q.enqueue(&q,T->rightChild);
        q.dequeue(&q);
    }
}
