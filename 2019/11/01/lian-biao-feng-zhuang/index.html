链表基本介绍

我们都学过链表,单链表,循环链表,双向链表. 有没有想过把这些链表操作全部结合到一起呢?

把他们的函数调用像调用成员那样,直接用.就可以操作.

废话不多说,由于C语言不像C++可以直接在结构体struct定义函数成员,但是C语言精髓指针操作可以做到. 下面我们利用C语言的函数指针达到封装效果.

定义链表

  1. 首先我们选创建一个LinkedList.h头文件

    #ifndef DATASTRUCT_LINKEDLIST_H
#define DATASTRUCT_LINKEDLIST_H //定义宏,防止编译器重复编译

//在宏中间编写代码
#include <stdlib.h>   //申请节点,需要用的头文件

#define true  (char)1 //由于c没有bool型,我们自定义bool
#define false (char)0
typedef char Status;

typedef void DataType;//抽象数据类型

#endif//

  2. 定义链表节点结构体,就是双向链表节点的结构体

    typedef struct LinkedListNode{

   struct LinkedListNode *previous;
   struct LinkedListNode *next;
   DataType *dataType;   //抽象数据
  }LinkedListNode;

  3. 定义链表结构体

    想想需要哪些操作?

    • 链表长度: 我们把length定义int;
    • 获取节点: 我们把函数分成三个,一个头,一个尾,最后一个应该能随便访问任何一个位置
    • 增加节点: 同上面一样的操作,只不过我把随便增加节点函数命名插入insert
    • 删除节点: 同上, 头尾移除命名为remove,能随便删除命为delete
    • 遍历: 我们创建了链表,肯定得瞅一瞅,不过这个得结合printData()这个函数
    • 排序: 我们偶尔也要排排成绩嘛,:raised_hand_with_fingers_splayed: 我成绩一般都是前五(倒序排)
    • 清空: 链表数据没有用了我们就把他清空掉
    • printData: 定义打印数据的接口,学生,教师,图书,物品每一种数据结构都不一样,这个需要重新定义
    • freeData: 当你需要删除动态数据时候你需要重新定义,默认空函数;
    typedef struct LinkedList{
    LinkedListNode *head;
    int length;
    DataType* (*get)(struct LinkedList *L,int index);
    DataType* (*getHead)(struct LinkedList *L);
    DataType* (*getTail)(struct LinkedList *L);
    Status (*addHead)(struct LinkedList *L,DataType *dataType);
    Status (*addTail)(struct LinkedList *L,DataType *dataType);
    Status (*removeHead)(struct LinkedList *L);
    Status (*removeTail)(struct LinkedList *L);
    Status (*insert)(struct LinkedList *L,DataType *dataType,int index);
    Status (*delete)(struct LinkedList *L,int index);
    Status (*traversal)(struct LinkedList *L);
    Status (*sort)(struct LinkedList *L);
    Status (*clean)(struct LinkedList *L);
    Status (*printData)(DataType *dataType);
    Status (*freeData)(DataType *dataType);
}LinkedList;

  4. 结构体定义完了,我们需要初始化这些数据,接下来定义初始化函数(类似new这个操作,只不过我们是静态分配,我们也可以写成动态定义)

     LinkedList initLinkedList();

定义函数

  1. 我们创建一个LinkedList.c 文件

    #include "LinkedList.h"//这个必须包含进去

  2. *相信你们都熟悉链表操作函数,在这里我要谈一下,咱们得把函数设置成static这个主要屏蔽外部文件访问这个函数,主要还是避免日后写了个重复函数名 *

    对了,前面忘说了,我们把这个链表设置有头节点的,方便操作.

获取节点函数

   static DataType *get(LinkedList *L,int index)
   {
       if(L->length==0||index>L->length||index<1)
           return NULL;
       int pos=1;
       LinkedListNode *p=L->head->next;
       while (p!=L->head&&pos<index)
       {
           p=p->next;
           ++pos;
       }
       return p->dataType;
   }
   static DataType *getHead(LinkedList *L) {
       if (L->length==0)
           return NULL;
       return L->head->next->dataType;
   }

   static DataType *getTail(LinkedList *L) {
       if(L->length==0)
           return NULL;
       return L->head->previous->dataType;
   }

头尾加函数

利用循环链表的特性,直接添加O(1)

static Status addHead(LinkedList *L,DataType *dataType){
    LinkedListNode *new=(LinkedListNode*)malloc(sizeof(LinkedListNode));
    if (new==NULL)
        return false;
    new->dataType=dataType;
    new->next=L->head->next;
    new->previous=L->head;
    L->head->next->previous=new;
    L->head->next=new;
    ++L->length;
    return true;
}

static Status addTail(LinkedList *L,DataType *dataType){
    LinkedListNode *new=(LinkedListNode*)malloc(sizeof(LinkedListNode));
    if (new==NULL)
        return false;
    new->dataType=dataType;
    new->next=L->head;
    new->previous=L->head->previous;
    L->head->previous->next=new;
    L->head->previous=new;
    ++L->length;
    return true;
}

头尾删除函数

利用循环链表的特性,直接删除O(1)

static Status removeHead(struct LinkedList *L) {
    if (L->length==0)
        return false;
    LinkedListNode *p=L->head->next;
    L->head->next=p->next;
    p->next->previous=L->head;
    L->freeData(p->dataType);
    free(p);
    --L->length;
    return true;
}

static Status removeTail(struct LinkedList *L) {
    if (L->length==0)
        return false;
    LinkedListNode *p=L->head->previous;
    L->head->previous=p->previous;
    p->previous->next=L->head;
    L->freeData(p->dataType);
    free(p);
    --L->length;
    return true;
}

随机删除或者插入

我设计的是从头到尾顺序查找插入or删除,O(n),不过还可以优化判断一下从头还是从尾开始,这样可以减少循环次数.

static Status insert(struct LinkedList *L,DataType *dataType,int index) {
    if (index<1||index>L->length+1)
        return false;
    //如果是头尾直接跳用
    if (index==1||index==L->length+1)
    {
        if(index==1)
            L->addHead(L,dataType);
        else
            L->addTail(L,dataType);
        return true;
    }
    LinkedListNode *new=(LinkedListNode*)malloc(sizeof(LinkedListNode));
    if (new==NULL)
        return false;
    LinkedListNode *p=L->head;
    int pos=1;
    while (p->next!=L->head&&pos<index)
    {
        p=p->next;
        ++pos;
    }
    new->dataType=dataType;
    new->next=p->next;
    new->previous=p;
    p->next->previous=new;
    p->next=new;
    ++L->length;
    return true;
}

static Status delete(struct LinkedList *L,int index) {
    if (index<1||index>L->length)
        return false;
    //如果是头尾直接跳用
    if (index==1||index==L->length)
    {
        if(index==1)
            L->removeHead(L);
        else
            L->removeTail(L);
        return true;
    }
    LinkedListNode *p=L->head->next;
    int pos=1;
    while (p->next!=L->head&&pos<index)
    {
        p=p->next;
        ++pos;
    }
    p->next->previous=p->previous;
    p->previous->next=p->next;
    L->freeData(p->dataType);
    free(p);
    --L->length;
    return true;
}

static Status clean(struct LinkedList *L) {
    if (L->length==0)
        return false;

}

initLinked List()

这个函数非常重要,c语言所有结构体成员都需要赋值才能进行各种各样的操作,LinkedList()函数就不需要加关键字static了,初始函数是提供给其他文件访问的.

LinkedList initLinkedList()
{
    LinkedList L;
    L.head=(LinkedListNode*)malloc(sizeof(LinkedListNode));
    L.head->dataType=NULL;
    L.head->next=L.head;
    L.head->previous=L.head;
    L.length=0;
    L.get=get;
    L.getHead=getHead;
    L.getTail=getTail;
    L.addHead=addHead;
    L.addTail=addTail;
    L.removeHead=removeHead;
    L.removeTail=removeTail;
    L.insert=insert;
    L.delete=delete;
    L.traversal=traversal;
    L.sort=sort;
    L.clean=clean;
    L.printData=printDataType;
    L.freeData=freeData;
    return L;
}
